17 //
18 // An *implicit* representation of the BFGS approximation to the Hessian matrix B
19 //
20 // B = theta * I - W * M * W'
21 // H = inv(B)
22 //
23 // Reference:
24 // [1] D. C. Liu and J. Nocedal (1989). On the limited memory BFGS method for large scale optimization.
25 // [2] R. H. Byrd, P. Lu, and J. Nocedal (1995). A limited memory algorithm for bound constrained optimization.
26 //
27 template <typename Scalar, bool LBFGSB = false>
28 class BFGSMat
29 {
30 private:
31  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;
32  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Matrix;
33  typedef Eigen::Ref<const Vector> RefConstVec;
34  typedef std::vector<int> IndexSet;
35 
36  int m_m; // Maximum number of correction vectors
37  Scalar m_theta; // theta * I is the initial approximation to the Hessian matrix
38  Matrix m_s; // History of the s vectors
39  Matrix m_y; // History of the y vectors
40  Vector m_ys; // History of the s'y values
41  Vector m_alpha; // Temporary values used in computing H * v
42  int m_ncorr; // Number of correction vectors in the history, m_ncorr <= m
43  int m_ptr; // A Pointer to locate the most recent history, 1 <= m_ptr <= m
44  // Details: s and y vectors are stored in cyclic order.
45  // For example, if the current s-vector is stored in m_s[, m-1],
46  // then in the next iteration m_s[, 0] will be overwritten.
47  // m_s[, m_ptr-1] points to the most recent history,
48  // and m_s[, m_ptr % m] points to the most distant one.
49 
50  //========== The following members are only used in L-BFGS-B algorithm ==========//
51  Matrix m_permMinv; // Permutated M inverse
52  BKLDLT<Scalar> m_permMsolver; // Represents the permutated M matrix
53 
54 public:
55  // Constructor
56  BFGSMat() {}

58  // Reset internal variables
59  // n: dimension of the vector to be optimized
60  // m: maximum number of corrections to approximate the Hessian matrix
61  inline void reset(int n, int m)
62  {
63  m_m = m;
64  m_theta = Scalar(1);
65  m_s.resize(n, m);
66  m_y.resize(n, m);
67  m_ys.resize(m);
68  m_alpha.resize(m);
69  m_ncorr = 0;
70  m_ptr = m; // This makes sure that m_ptr % m == 0 in the first step
71 
72  if(LBFGSB)
73  {
74  m_permMinv.resize(2 * m, 2 * m);
75  m_permMinv.diagonal().setOnes();
76  }
77  }
78 
79  // Add correction vectors to the BFGS matrix
80  inline void add_correction(const RefConstVec& s, const RefConstVec& y)
81  {
82  const int loc = m_ptr % m_m;
83 
84  m_s.col(loc).noalias() = s;
85  m_y.col(loc).noalias() = y;
86 
87  // ys = y's = 1/rho
88  const Scalar ys = m_s.col(loc).dot(m_y.col(loc));
89  m_ys[loc] = ys;
90 
91  m_theta = m_y.col(loc).squaredNorm() / ys;
92 
93  if(m_ncorr < m_m)
94  m_ncorr++;
95 
96  m_ptr = loc + 1;
97 
98  if(LBFGSB)
99  {
100  // Minv = [-D L']
101  // [ L theta*S'S]
102 
103  // Copy -D
104  // Let S=[s[0], ..., s[m-1]], Y=[y[0], ..., y[m-1]]
105  // D = [s[0]'y[0], ..., s[m-1]'y[m-1]]
106  m_permMinv(loc, loc) = -ys;
107 
108  // Update S'S
109  // We only store S'S in Minv, and multiply theta when LU decomposition is performed
110  Vector Ss = m_s.leftCols(m_ncorr).transpose() * m_s.col(loc);
111  m_permMinv.block(m_m + loc, m_m, 1, m_ncorr).noalias() = Ss.transpose();
112  m_permMinv.block(m_m, m_m + loc, m_ncorr, 1).noalias() = Ss;
113 
114  // Compute L
115  // L = [ 0 ]
116  // [ s[1]'y[0] 0 ]
117  // [ s[2]'y[0] s[2]'y[1] ]
118  // ...
119  // [s[m-1]'y[0] ... ... ... ... ... s[m-1]'y[m-2] 0]
120  //
121  // L_next = [ 0 ]
122  // [s[2]'y[1] 0 ]
123  // [s[3]'y[1] s[3]'y[2] ]
124  // ...
125  // [s[m]'y[1] ... ... ... ... ... s[m]'y[m-1] 0]
126  const int len = m_ncorr - 1;
127  // First zero out the column of oldest y
128  if(m_ncorr >= m_m)
129  m_permMinv.block(m_m, loc, m_m, 1).setZero();
130  // Compute the row associated with new s
131  // The current row is loc
132  // End with column (loc + m - 1) % m
133  // Length is len
134  int yloc = (loc + m_m - 1) % m_m;
135  for(int i = 0; i < len; i++)
136  {
137  m_permMinv(m_m + loc, yloc) = m_s.col(loc).dot(m_y.col(yloc));
138  yloc = (yloc + m_m - 1) % m_m;
139  }
140 
141  // Matrix LDLT factorization
142  m_permMinv.block(m_m, m_m, m_m, m_m) *= m_theta;
143  m_permMsolver.compute(m_permMinv);
144  m_permMinv.block(m_m, m_m, m_m, m_m) /= m_theta;
145  }
146  }
147 
148  // Recursive formula to compute a * H * v, where a is a scalar, and v is [n x 1]
149  // H0 = (1/theta) * I is the initial approximation to H
150  // Algorithm 7.4 of Nocedal, J., & Wright, S. (2006). Numerical optimization.
151  inline void apply_Hv(const Vector& v, const Scalar& a, Vector& res)
152  {
153  res.resize(v.size());
154 
155  // L-BFGS two-loop recursion
156 
157  // Loop 1
158  res.noalias() = a * v;
159  int j = m_ptr % m_m;
160  for(int i = 0; i < m_ncorr; i++)
161  {
162  j = (j + m_m - 1) % m_m;
163  m_alpha[j] = m_s.col(j).dot(res) / m_ys[j];
164  res.noalias() -= m_alpha[j] * m_y.col(j);
165  }
166 
167  // Apply initial H0
168  res /= m_theta;
169 
170  // Loop 2
171  for(int i = 0; i < m_ncorr; i++)
172  {
173  const Scalar beta = m_y.col(j).dot(res) / m_ys[j];
174  res.noalias() += (m_alpha[j] - beta) * m_s.col(j);
175  j = (j + 1) % m_m;
176  }
177  }

179  //========== The following functions are only used in L-BFGS-B algorithm ==========//

181  // Return the value of theta
182  inline Scalar theta() const { return m_theta; }
183 
184  // Return current number of correction vectors
185  inline int num_corrections() const { return m_ncorr; }

187  // W = [Y, theta * S]
188  // W [n x (2*ncorr)], v [n x 1], res [(2*ncorr) x 1]
189  // res preserves the ordering of Y and S columns
190  inline void apply_Wtv(const Vector& v, Vector& res) const
191  {
192  res.resize(2 * m_ncorr);
193  res.head(m_ncorr).noalias() = m_y.leftCols(m_ncorr).transpose() * v;
194  res.tail(m_ncorr).noalias() = m_theta * m_s.leftCols(m_ncorr).transpose() * v;
195  }
196 
197  // The b-th row of the W matrix
198  // Preserves the ordering of Y and S columns
199  // Return as a column vector
200  inline Vector Wb(int b) const
201  {
202  Vector res(2 * m_ncorr);
203  for(int j = 0; j < m_ncorr; j++)
204  {
205  res[j] = m_y(b, j);
206  res[m_ncorr + j] = m_s(b, j);
207  }
208  res.tail(m_ncorr) *= m_theta;
209  return res;
210  }
211 
212  // Extract rows of W
213  inline Matrix Wb(const IndexSet& b) const
214  {
215  const int nb = b.size();
216  const int* bptr = b.data();
217  Matrix res(nb, 2 * m_ncorr);
218 
219  for(int j = 0; j < m_ncorr; j++)
220  {
221  const Scalar* Yptr = &m_y(0, j);
222  const Scalar* Sptr = &m_s(0, j);
223  Scalar* resYptr = res.data() + j * nb;
224  Scalar* resSptr = resYptr + m_ncorr * nb;
225  for(int i = 0; i < nb; i++)
226  {
227  const int row = bptr[i];
228  resYptr[i] = Yptr[row];
229  resSptr[i] = Sptr[row];
230  }
231  }
232  return res;
233  }

235  // M is [(2*ncorr) x (2*ncorr)], v is [(2*ncorr) x 1]
236  inline void apply_Mv(const Vector& v, Vector& res) const
237  {
238  res.resize(2 * m_ncorr);
239  if(m_ncorr < 1)
240  return;
241 
242  Vector vpadding = Vector::Zero(2 * m_m);
243  vpadding.head(m_ncorr).noalias() = v.head(m_ncorr);
244  vpadding.segment(m_m, m_ncorr).noalias() = v.tail(m_ncorr);
245 
246  // Solve linear equation
247  m_permMsolver.solve_inplace(vpadding);
248 
249  res.head(m_ncorr).noalias() = vpadding.head(m_ncorr);
250  res.tail(m_ncorr).noalias() = vpadding.segment(m_m, m_ncorr);
251  }
252 
253  // Compute W'Pv
254  // W [n x (2*ncorr)], v [nP x 1], res [(2*ncorr) x 1]
255  // res preserves the ordering of Y and S columns
256  // Returns false if the result is known to be zero
257  inline bool apply_WtPv(const IndexSet& P_set, const Vector& v, Vector& res, bool test_zero = false) const
258  {
259  const int* Pptr = P_set.data();
260  const Scalar* vptr = v.data();
261  int nP = P_set.size();
262 
263  // Remove zeros in v to save computation
264  IndexSet P_reduced;
265  std::vector<Scalar> v_reduced;
266  if(test_zero)
267  {
268  P_reduced.reserve(nP);
269  for(int i = 0; i < nP; i++)
270  {
271  if(vptr[i] != Scalar(0))
272  {
273  P_reduced.push_back(Pptr[i]);
274  v_reduced.push_back(vptr[i]);
275  }
276  }
277  Pptr = P_reduced.data();
278  vptr = v_reduced.data();
279  nP = P_reduced.size();
280  }
281 
282  res.resize(2 * m_ncorr);
283  if(m_ncorr < 1 || nP < 1)
284  {
285  res.setZero();
286  return false;
287  }
288 
289  for(int j = 0; j < m_ncorr; j++)
290  {
291  Scalar resy = Scalar(0), ress = Scalar(0);
292  const Scalar* yptr = &m_y(0, j);
293  const Scalar* sptr = &m_s(0, j);
294  for(int i = 0; i < nP; i++)
295  {
296  const int row = Pptr[i];
297  resy += yptr[row] * vptr[i];
298  ress += sptr[row] * vptr[i];
299  }
300  res[j] = resy;
301  res[m_ncorr + j] = ress;
302  }
303  res.tail(m_ncorr) *= m_theta;
304  return true;
305  }

307  // Compute s * P'WMv
308  // Assume that v[2*ncorr x 1] has the same ordering (permutation) as W and M
309  // Returns false if the result is known to be zero
310  inline bool apply_PtWMv(const IndexSet& P_set, const Vector& v, Vector& res, const Scalar& scale) const
311  {
312  const int nP = P_set.size();
313  res.resize(nP);
314  res.setZero();
315  if(m_ncorr < 1 || nP < 1)
316  return false;
317 
318  Vector Mv;
319  apply_Mv(v, Mv);
320  // WP * Mv
321  Mv.tail(m_ncorr) *= m_theta;
322  for(int j = 0; j < m_ncorr; j++)
323  {
324  const Scalar* yptr = &m_y(0, j);
325  const Scalar* sptr = &m_s(0, j);
326  const Scalar Mvy = Mv[j], Mvs = Mv[m_ncorr + j];
327  for(int i = 0; i < nP; i++)
328  {
329  const int row = P_set[i];
330  res[i] += Mvy * yptr[row] + Mvs * sptr[row];
331  }
332  }
333  res *= scale;
334  return true;
335  }
336  // If the P'W matrix has been explicitly formed, do a direct matrix multiplication
337  inline bool apply_PtWMv(const Matrix& WP, const Vector& v, Vector& res, const Scalar& scale) const
338  {
339  const int nP = WP.rows();
340  res.resize(nP);
341  if(m_ncorr < 1 || nP < 1)
342  {
343  res.setZero();
344  return false;
345  }
346 
347  Vector Mv;
348  apply_Mv(v, Mv);
349  // WP * Mv
350  Mv.tail(m_ncorr) *= m_theta;
351  res.noalias() = scale * (WP * Mv);
352  return true;
353  }

355  // Compute F'BAb = -(F'W)M(W'AA'd)
356  // W'd is known, and AA'+FF'=I, so W'AA'd = W'd - W'FF'd
357  // Usually d contains many zeros, so we fist compute number of nonzero elements in A set and F set,
358  // denoted as nnz_act and nnz_fv, respectively
359  // If nnz_act is smaller, compute W'AA'd = WA' (A'd) directly
360  // If nnz_fv is smaller, compute W'AA'd = W'd - WF' * (F'd)
361  inline void compute_FtBAb(
362  const IndexSet& fv_set, const IndexSet& newact_set, const Vector& Wd, const Vector& drt, Vector& res
363  ) const
364  {
365  const int nact = newact_set.size();
366  const int nfree = fv_set.size();
367  res.resize(nfree);
368  if(m_ncorr < 1 || nact < 1 || nfree < 1)
369  {
370  res.setZero();
371  return;
372  }
373 
374  // W'AA'd
375  Vector rhs(2 * m_ncorr);
376  if(nact <= nfree)
377  {
378  // Construct A'd
379  Vector Ad(nfree);
380  for(int i = 0; i < nact; i++)
381  Ad[i] = drt[newact_set[i]];
382  apply_WtPv(newact_set, Ad, rhs);
383  } else {
384  // Construct F'd
385  Vector Fd(nfree);
386  for(int i = 0; i < nfree; i++)
387  Fd[i] = drt[fv_set[i]];
388  // Compute W'AA'd = W'd - WF' * (F'd)
389  apply_WtPv(fv_set, Fd, rhs);
390  rhs.noalias() = Wd - rhs;
391  }
392 
393  apply_PtWMv(fv_set, rhs, res, Scalar(-1));
394  }
395 
396  // Compute inv(P'BP) * v
397  // P represents an index set
398  // inv(P'BP) * v = v / theta + WP * inv(inv(M) - WP' * WP / theta) * WP' * v / theta^2
399  //
400  // v is [nP x 1]
401  inline void solve_PtBP(const Matrix& WP, const Vector& v, Vector& res) const
402  {
403  const int nP = WP.rows();
404  res.resize(nP);
405  if(m_ncorr < 1 || nP < 1)
406  {
407  res.noalias() = v / m_theta;
408  return;
409  }
410 
411  // Compute the matrix in the middle (only the lower triangular part is needed)
412  // Remember that W = [Y, theta * S], but we do not store theta in WP
413  Matrix mid(2 * m_ncorr, 2 * m_ncorr);
414  // [0:(ncorr - 1), 0:(ncorr - 1)]
415  for(int j = 0; j < m_ncorr; j++)
416  {
417  mid.col(j).segment(j, m_ncorr - j).noalias() = m_permMinv.col(j).segment(j, m_ncorr - j) -
418  WP.block(0, j, nP, m_ncorr - j).transpose() * WP.col(j) / m_theta;
419  }
420  // [ncorr:(2 * ncorr - 1), 0:(ncorr - 1)]
421  mid.block(m_ncorr, 0, m_ncorr, m_ncorr).noalias() = m_permMinv.block(m_m, 0, m_ncorr, m_ncorr) -
422  WP.rightCols(m_ncorr).transpose() * WP.leftCols(m_ncorr);
423  // [ncorr:(2 * ncorr - 1), ncorr:(2 * ncorr - 1)]
424  for(int j = 0; j < m_ncorr; j++)
425  {
426  mid.col(m_ncorr + j).segment(m_ncorr + j, m_ncorr - j).noalias() = m_theta *
427  (m_permMinv.col(m_m + j).segment(m_m + j, m_ncorr - j) - WP.rightCols(m_ncorr - j).transpose() * WP.col(m_ncorr + j));
428  }
429  // Factorization
430  BKLDLT<Scalar> midsolver(mid);
431  // Compute the final result
432  Vector WPv = WP.transpose() * v;
433  WPv.tail(m_ncorr) *= m_theta;
434  midsolver.solve_inplace(WPv);
435  WPv.tail(m_ncorr) *= m_theta;
436  res.noalias() = v / m_theta + (WP * WPv) / (m_theta * m_theta);
437  }

439  // Compute P'BQv, where P and Q are two mutually exclusive index selection operators
440  // P'BQv = -WP * M * WQ' * v
441  // Returns false if the result is known to be zero
442  inline bool apply_PtBQv(const Matrix& WP, const IndexSet& Q_set, const Vector& v, Vector& res, bool test_zero = false) const
443  {
444  const int nP = WP.rows();
445  const int nQ = Q_set.size();
446  res.resize(nP);
447  if(m_ncorr < 1 || nP < 1 || nQ < 1)
448  {
449  res.setZero();
450  return false;
451  }
452 
453  Vector WQtv;
454  bool nonzero = apply_WtPv(Q_set, v, WQtv, test_zero);
455  if(!nonzero)
456  {
457  res.setZero();
458  return false;
459  }
460 
461  Vector MWQtv;
462  apply_Mv(WQtv, MWQtv);
463  MWQtv.tail(m_ncorr) *= m_theta;
464  res.noalias() = -WP * MWQtv;
465  return true;
466  }
467  // If the Q'W matrix has been explicitly formed, do a direct matrix multiplication
468  inline bool apply_PtBQv(const Matrix& WP, const Matrix& WQ, const Vector& v, Vector& res) const
469  {
470  const int nP = WP.rows();
471  const int nQ = WQ.rows();
472  res.resize(nP);
473  if(m_ncorr < 1 || nP < 1 || nQ < 1)
474  {
475  res.setZero();
476  return false;
477  }
478 
479  // Remember that W = [Y, theta * S], so we need to multiply theta to the second half
480  Vector WQtv = WQ.transpose() * v;
481  WQtv.tail(m_ncorr) *= m_theta;
482  Vector MWQtv;
483  apply_Mv(WQtv, MWQtv);
484  MWQtv.tail(m_ncorr) *= m_theta;
485  res.noalias() = -WP * MWQtv;
486  return true;
487  }
