46 template <typename Foo>
47 static void LineSearch(Foo& f, const LBFGSParam<Scalar>& param,
48 const Vector& xp, const Vector& drt, const Scalar& step_max,
49 Scalar& step, Scalar& fx, Vector& grad, Scalar& dg, Vector& x)
50 {
51 // Check the value of step
52 if (step <= Scalar(0))
53 throw std::invalid_argument("'step' must be positive");
54
55 // Save the function value at the current x
56 const Scalar fx_init = fx;
57 // Projection of gradient on the search direction
58 const Scalar dg_init = grad.dot(drt);
59 // Make sure d points to a descent direction
60 if (dg_init > 0)
61 throw std::logic_error("the moving direction increases the objective function value");
62
63 const Scalar test_decr = param.ftol * dg_init;
64
65 // Upper and lower end of the current line search range
66 Scalar step_lo = 0,
67 step_hi = std::numeric_limits<Scalar>::infinity();
68
69 int iter;
70 for (iter = 0; iter < param.max_linesearch; iter++)
71 {
72 // x_{k+1} = x_k + step * d_k
73 x.noalias() = xp + step * drt;
74 // Evaluate this candidate
75 fx = f(x, grad);
76
77 if (fx > fx_init + step * test_decr || (fx != fx))
78 {
79 step_hi = step;
80 }
81 else
82 {
83 // Armijo condition is met
84 if (param.linesearch == LBFGS_LINESEARCH_BACKTRACKING_ARMIJO)
85 break;
86
87 const Scalar dg = grad.dot(drt);
88 if (dg < param.wolfe * dg_init)
89 {
90 step_lo = step;
91 }
92 else
93 {
94 // Regular Wolfe condition is met
95 if (param.linesearch == LBFGS_LINESEARCH_BACKTRACKING_WOLFE)
96 break;
97
98 if (dg > -param.wolfe * dg_init)
99 {
100 step_hi = step;
101 }
102 else
103 {
104 // Strong Wolfe condition is met
105 break;
106 }
107 }
108 }
109
110 assert(step_lo < step_hi);
111
112 if (step < param.min_step)
113 throw std::runtime_error("the line search step became smaller than the minimum value allowed");
114
115 if (step > param.max_step)
116 throw std::runtime_error("the line search step became larger than the maximum value allowed");
117
118 // continue search in mid of current search range
119 step = std::isinf(step_hi) ? 2 * step : step_lo / 2 + step_hi / 2;
120 }
121
122 if (iter >= param.max_linesearch)
123 throw std::runtime_error("the line search routine reached the maximum number of iterations");
124 }
125};
