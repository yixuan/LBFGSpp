20template <typename Scalar>
21class LineSearchNocedalWright
22{
23private:
24 typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;
25
26public:
48 template <typename Foo>
49 static void LineSearch(Foo& f, const LBFGSParam<Scalar>& param,
50 const Vector& xp, const Vector& drt, const Scalar& step_max,
51 Scalar& step, Scalar& fx, Vector& grad, Scalar& dg, Vector& x)
52 {
53 // Check the value of step
54 if (step <= Scalar(0))
55 throw std::invalid_argument("'step' must be positive");
56
57 if (param.linesearch != LBFGS_LINESEARCH_BACKTRACKING_STRONG_WOLFE)
58 throw std::invalid_argument("'param.linesearch' must be 'LBFGS_LINESEARCH_BACKTRACKING_STRONG_WOLFE' for LineSearchNocedalWright");
59
60 // To make this implementation more similar to the other line search
61 // methods in LBFGSpp, the symbol names from the literature
62 // ("Numerical Optimizations") have been changed.
63 //
64 // Literature | LBFGSpp
65 // -----------|--------
66 // alpha | step
67 // phi | fx
68 // phi' | dg
69
70 // The expansion rate of the step size
71 const Scalar expansion = Scalar(2);
72
73 // Save the function value at the current x
74 const Scalar fx_init = fx;
75 // Projection of gradient on the search direction
76 const Scalar dg_init = grad.dot(drt);
77 // Make sure d points to a descent direction
78 if (dg_init > Scalar(0))
79 throw std::logic_error("the moving direction increases the objective function value");
80
81 const Scalar test_decr = param.ftol * dg_init, // Sufficient decrease
82 test_curv = -param.wolfe * dg_init; // Curvature
83
84 // Ends of the line search range (step_lo > step_hi is allowed)
85 Scalar step_hi, fx_hi, dg_hi;
86 Scalar step_lo = Scalar(0), fx_lo = fx_init, dg_lo = dg_init;
87
88 // STEP 1: Bracketing Phase
89 // Find a range guaranteed to contain a step satisfying strong Wolfe.
90 //
91 // See also:
92 // "Numerical Optimization", "Algorithm 3.5 (Line Search Algorithm)".
93 int iter = 0;
94 for (;;)
95 {
96 // Evaluate the current step size
97 x.noalias() = xp + step * drt;
98 fx = f(x, grad);
99
100 iter++;
101 if (iter >= param.max_linesearch)
102 throw std::runtime_error("the line search routine reached the maximum number of iterations");
103
104 const Scalar dg = grad.dot(drt);
105
106 // Test the sufficient decrease condition
107 if (fx - fx_init > step * test_decr || (Scalar(0) < step_lo && fx >= fx_lo))
108 {
109 step_hi = step;
110 fx_hi = fx;
111 dg_hi = dg;
112 break;
113 }
114 // If reaching here, then the sufficient decrease condition is satisfied
115
116 // Test the curvature condition
117 if (std::abs(dg) <= test_curv)
118 return;
119
120 step_hi = step_lo;
121 fx_hi = fx_lo;
122 dg_hi = dg_lo;
123 step_lo = step;
124 fx_lo = fx;
125 dg_lo = dg;
126
127 if (dg >= Scalar(0))
128 break;
129
130 step *= expansion;
131 }
132
133 // STEP 2: Zoom Phase
134 // Given a range (step_lo,step_hi) that is guaranteed to
135 // contain a valid strong Wolfe step value, this method
136 // finds such a value.
137 //
138 // See also:
139 // "Numerical Optimization", "Algorithm 3.6 (Zoom)".
140 for (;;)
141 {
142 // Use {fx_lo, fx_hi, dg_lo} to make a quadratic interpolation of
143 // the function, and the fitted quadratic function is used to
144 // estimate the minimum
145 //
146 // polynomial: p (x) = c0*(x - step)² + c1
147 // conditions: p (step_hi) = fx_hi
148 // p (step_lo) = fx_lo
149 // p'(step_lo) = dg_lo
150
151 // We allow fx_hi to be Inf, so first compute a candidate for step size,
152 // and test whether NaN occurs
153 const Scalar fdiff = fx_hi - fx_lo;
154 const Scalar sdiff = step_hi - step_lo;
155 const Scalar smid = (step_hi + step_lo) / Scalar(2);
156 Scalar step_candid = fdiff * step_lo - smid * sdiff * dg_lo;
157 step_candid = step_candid / (fdiff - sdiff * dg_lo);
158
159 // In some cases the interpolation is not a good choice
160 // This includes (a) NaN values; (b) too close to the end points; (c) outside the interval
161 // In such cases, a bisection search is used
162 const bool candid_nan = !(std::isfinite(step_candid));
163 const Scalar end_dist = std::min(std::abs(step_candid - step_lo), std::abs(step_candid - step_hi));
164 const bool near_end = end_dist < Scalar(0.01) * std::abs(sdiff);
165 const bool bisect = candid_nan ||
166 (step_candid <= std::min(step_lo, step_hi)) ||
167 (step_candid >= std::max(step_lo, step_hi)) ||
168 near_end;
169 step = bisect ? smid : step_candid;
170
171 // Evaluate the current step size
172 x.noalias() = xp + step * drt;
173 fx = f(x, grad);
174
175 iter++;
176 if (iter >= param.max_linesearch)
177 throw std::runtime_error("the line search routine reached the maximum number of iterations");
178
179 const Scalar dg = grad.dot(drt);
180
181 // Test the sufficient decrease condition
182 if (fx - fx_init > step * test_decr || fx >= fx_lo)
183 {
184 if (step == step_hi)
185 throw std::runtime_error("the line search routine failed, possibly due to insufficient numeric precision");
186
187 step_hi = step;
188 fx_hi = fx;
189 dg_hi = dg;
190 }
191 else
192 {
193 // Test the curvature condition
194 if (std::abs(dg) <= test_curv)
195 return;
196
197 if (dg * (step_hi - step_lo) >= Scalar(0))
198 {
199 step_hi = step_lo;
200 fx_hi = fx_lo;
201 dg_hi = dg_lo;
202 }
203
204 if (step == step_lo)
205 throw std::runtime_error("the line search routine failed, possibly due to insufficient numeric precision");
206
207 step_lo = step;
208 fx_lo = fx;
209 dg_lo = dg;
210 }
211 }
212 }
213};
